-- Crear la base de datos 'example1'
create database example1;

-- Crear la tabla 'supplier' para almacenar información de proveedores
create table supplier
(
    id int not null auto_increment primary key, -- Clave primaria autoincremental como identificador único
    name varchar(100), -- Nombre del proveedor
    address varchar(200), -- Dirección del proveedor
    phone int -- Teléfono del proveedor
);

-- Crear la tabla 'product' para almacenar información de productos
create table product
(
    id int not null auto_increment primary key, -- Clave primaria autoincremental como identificador único
    name varchar(100), -- Nombre del producto
    price int, -- Precio del producto
    quantity varchar(30) -- Cantidad disponible del producto
);
